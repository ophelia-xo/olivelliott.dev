// app/layout.tsx — RSC (no 'use client')
import type { Metadata } from 'next'
import { GeistSans } from 'geist/font/sans'
import { GeistMono } from 'geist/font/mono'
import { Providers } from './providers'
import './globals.css'

export const metadata: Metadata = {
  metadataBase: new URL(process.env.NEXT_PUBLIC_SITE_URL ?? 'https://olivelliott.dev'),
  title: { default: 'olivelliott.dev', template: '%s · olivelliott.dev' },
  description: 'Olive Elliott — engineer building tools for autonomy, local-first systems, and open-source communities.',
}

export default function RootLayout({ children }: { children: React.ReactNode }) {
  return (
    <html lang="en" className={`${GeistSans.variable} ${GeistMono.variable}`} suppressHydrationWarning>
      <body className="bg-bg text-text-primary font-sans antialiased">
        <Providers>{children}</Providers>
      </body>
    </html>
  )
}

// app/providers.tsx — 'use client'
'use client'
import { ThemeProvider } from 'next-themes'

export function Providers({ children }: { children: React.ReactNode }) {
  return (
    <ThemeProvider attribute="class" defaultTheme="dark" enableSystem={false} disableTransitionOnChange>
      {children}
    </ThemeProvider>
  )
}

// app/(site)/layout.tsx — RSC (MotionProvider/Nav/Footer added in Plan 04)
// Plan 02 ships a MINIMAL version with <main id="main"> so the skip-link target
// exists and the home page (Plan 05) can render. Plan 04 replaces this file
// in a coordinated update (files_modified overlap is intentional and tracked —
// Plan 04 depends on Plan 02 via Wave sequencing).
export default function SiteLayout({ children }: { children: React.ReactNode }) {
  return (
    <main id="main" className="mx-auto max-w-6xl px-6 md:px-8 lg:px-12 pt-8 md:pt-16 pb-16 md:pb-24">
      {children}
    </main>
  )
}

Notes the executor must enforce:
- `suppressHydrationWarning` ONLY on <html>. Not on <body>. Not on child elements. (Pitfall 6)
- Tailwind utility `bg-bg` resolves to `var(--color-bg)` via the @theme token layer from Plan 01. If `bg-bg` fails to render, verify `styles/tokens.css` has `--color-bg: #0a0a0a` and `app/globals.css` imports it.
- Tailwind utility `text-text-primary` may need alias adjustment depending on v4 token-name-to-utility mapping. If the utility does not resolve, use inline `style={{ color: 'var(--color-text-primary)' }}` as a fallback; the tokens are the source of truth, the utility name is a convenience.
