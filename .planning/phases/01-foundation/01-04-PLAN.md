// ----- components/site/nav-link.tsx — 'use client' -----
'use client'
import { usePathname } from 'next/navigation'
import { cn } from '@/lib/utils'

type NavLinkProps = { href: string; label: string }

export function NavLink({ href, label }: NavLinkProps): JSX.Element {
  const pathname = usePathname()
  const isActive = pathname === href || (href !== '/' && pathname.startsWith(href))
  return (
    <a
      href={href}
      aria-current={isActive ? 'page' : undefined}
      className={cn(
        'inline-block font-mono text-[0.875rem] font-medium tracking-wide lowercase px-3 py-3',
        'transition-colors duration-[120ms]',
        isActive
          ? 'text-text-primary border-b-[1px] border-accent pb-1'
          : 'text-text-secondary hover:text-text-primary',
      )}
    >
      {label}
    </a>
  )
}
