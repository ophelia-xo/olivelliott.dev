// vercel.json — optional but explicit
{
  "$schema": "https://openapi.vercel.sh/vercel.json",
  "framework": "nextjs",
  "buildCommand": "pnpm build",
  "installCommand": "pnpm install --frozen-lockfile",
  "outputDirectory": ".next",
  "cleanUrls": true
}

// package.json fields that Vercel reads automatically:
//   "packageManager": "pnpm@9.15.9"   → Vercel provisions pnpm 9.15.9
//   "engines": { "node": ">=20.18.0" } → Vercel provisions Node 20.x (or 22.x)

// Vercel deploy flow:
//   1. git push origin main
//   2. Vercel webhook fires
//   3. Vercel clones repo, reads packageManager + engines
//   4. corepack prepare pnpm@9.15.9
//   5. pnpm install --frozen-lockfile
//   6. pnpm build
//   7. Publish to production URL (olivelliott-dev.vercel.app or similar)

// NEXT_PUBLIC_SITE_URL env var:
//   - Local/default: https://olivelliott.dev (baked into metadataBase fallback)
//   - Vercel production: can be left unset (fallback used); or set to the Vercel URL
//   - Custom domain: deferred to Phase 7 per PROJECT.md
