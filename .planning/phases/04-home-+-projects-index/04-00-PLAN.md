    ```tsx
    'use client'

    import { m, useReducedMotion } from 'motion/react'
    import { useEffect, useState } from 'react'

    interface ThesisParagraphProps {
      text: string
      className?: string
    }

    /**
     * ThesisParagraph — the single earned motion moment of v1 (HOM-05).
     *
     * Renders the home-page thesis paragraph with a per-word opacity fade-in on mount.
     *
     * Correctness gates (load-bearing — see Phase 4 RESEARCH § Pitfalls 4 + 5):
     * - useReducedMotion() is explicitly checked. MotionConfig reducedMotion="user"
     *   does NOT collapse opacity animations — only transform/layout. We MUST gate
     *   manually or reduced-motion users get the per-word fade anyway.
     * - Two-stage mount pattern: useState(false) → useEffect setMounted(true).
     *   On the server and on the first client render, mounted === false → plain <p>
     *   matches SSR output exactly. No hydration mismatch. After mount the segmented
     *   form swaps in.
     * - Opacity-only animation, no transforms. CLS guard — the segmented and plain
     *   forms occupy the same layout box (inline spans, no padding, no transforms).
     * - LazyMotion strict (Phase 1 provider) requires m.* not motion.* — runtime
     *   throw otherwise.
     *
     * Phase 4 placement: composed by <HomeHero> (Plan 02). The thesis copy lives in
     * app/(site)/page.tsx (Plan 02) and is marked as a Phase 7 placeholder there.
     */
    export function ThesisParagraph({ text, className }: ThesisParagraphProps) {
      const [mounted, setMounted] = useState(false)
      const prefersReduced = useReducedMotion()

      useEffect(() => {
        setMounted(true)
      }, [])

      // SSR + pre-hydration + reduced-motion-on → render plain <p> at full opacity.
      // No flash-of-invisible-text; no flicker on reduced-motion toggle.
      if (!mounted || prefersReduced) {
        return <p className={className}>{text}</p>
      }

      // Motion permitted, post-hydration: per-word fade. Whitespace segments render
      // at opacity 1 immediately (no fade on whitespace) so the rhythm reads as
      // "words appearing", not "everything fading including the gaps".
      const segments = text.split(/(\s+)/)
      let wordIndex = 0
      return (
        <p className={className}>
          {segments.map((seg, i) => {
            if (/^\s+$/.test(seg)) {
              return (
                <span key={i} aria-hidden="true">
                  {seg}
                </span>
              )
            }
            const delay = wordIndex * 0.03
            wordIndex++
            return (
              <m.span
                key={i}
                initial={{ opacity: 0 }}
                animate={{ opacity: 1 }}
                transition={{ duration: 0.18, delay, ease: [0.22, 1, 0.36, 1] }}
              >
                {seg}
              </m.span>
            )
          })}
        </p>
      )
    }
    ```
