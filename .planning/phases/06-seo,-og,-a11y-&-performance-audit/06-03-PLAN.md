    // animate, initial, etc.) and forwards children to the underlying tag.

    vi.mock('motion/react', async () => {
      const React = await import('react')
      const motionTag = (Tag: string) =>
        ({ children, ...props }: Record<string, unknown>) => {
          const cleaned: Record<string, unknown> = {}

        m: new Proxy({}, { get: (_, tag: string) => motionTag(tag) }),
        motion: new Proxy({}, { get: (_, tag: string) => motionTag(tag) }),
        useReducedMotion: () => false,
        LazyMotion: ({ children }: { children: React.ReactNode }) => children,
        MotionConfig: ({ children }: { children: React.ReactNode }) => children,

        domAnimation: {},
      }

    import { render } from '@testing-library/react'

    const ROOT = path.resolve(__dirname, '..', '..')

    function walk(dir: string, exts = ['.ts', '.tsx']): string[] {
      const out: string[] = []
      for (const entry of readdirSync(dir)) {
        const abs = path.join(dir, entry)
        if (statSync(abs).isDirectory()) out.push(...walk(abs, exts))
        else if (exts.some((e) => abs.endsWith(e))) out.push(abs)
      }
      return out
    }

    function stripComments(src: string): string {
      return src
        .replace(/\/\*[\s\S]*?\*\//g, '')
        .split('\n')
        .map((l) => l.replace(/\/\/.*$/, ''))
        .join('\n')
